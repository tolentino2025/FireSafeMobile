Procurar se já existe um InspectionSchedule ativo com essa combinação de:

Se não existir:

criar um novo schedule:

startDate = inspection.date

isActive = true

Se já existir:

atualizar:

lastInspectionDate = inspection.date

nextDueDate = inspection.date + intervalo da frequência

Criar uma função utilitária para somar o intervalo:

function addInterval(date: Date, frequency: InspectionFrequency): Date {
  const d = new Date(date);

  switch (frequency) {
    case 'DAILY':
      d.setDate(d.getDate() + 1);
      break;
    case 'WEEKLY':
      d.setDate(d.getDate() + 7);
      break;
    case 'MONTHLY':
      d.setMonth(d.getMonth() + 1);
      break;
    case 'QUARTERLY':
      d.setMonth(d.getMonth() + 3);
      break;
    case 'YEARLY':
      d.setFullYear(d.getFullYear() + 1);
      break;
    case 'FIVE_YEAR': // testes de 5 anos da NFPA 25
      d.setFullYear(d.getFullYear() + 5);
      break;
  }

  return d;
}


IMPORTANTE: não alterar a forma como as inspeções são salvas hoje; apenas chamar createOrUpdateScheduleForInspection(inspection) logo após o salvamento de uma inspeção.
